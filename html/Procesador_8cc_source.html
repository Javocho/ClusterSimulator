33  else if (posicion.find(proc.get_id()) != posicion.end()) cout << "error: ya existe proceso" << endl;
34  else {

132 bool Procesador::hay_hueco_arriba(int pos, int& peso_hueco, int& pos_hueco) {
133  map<int, Proceso>::iterator it = procesos.find(pos);
134  if (it != procesos.begin()) {
135  --it;
136  int pos_nueva = it->first;
137  int peso_nuevo = it->second.get_mem();
138  if (peso_nuevo + pos_nueva == pos) return false; //proceso seguido del anterior (no hay hueco)
139  else {
140  peso_hueco = pos - (pos_nueva + peso_nuevo); //posición inicial proceso - posición final proceso anterior será el tamaño del hueco
141  pos_hueco = pos_nueva + peso_nuevo;
142  set<pair<int, int>, ComparePairs>::iterator it2 = huecos.find(make_pair(peso_hueco, pos_hueco));

144  return true;
145  }
146  }

148  else {
149  if (it->first == 0) return false;

152  pos_hueco = 0;

161  bool Procesador::hay_hueco_abajo(int pos, int& peso_hueco, int& pos_hueco) {
162  map<int, Proceso>::iterator it = procesos.find(pos);
163  int peso = it->second.get_mem();
164  ++it;
165  if (it != procesos.end()) {
166  if (pos + peso == it->first) return false; //si posición final del proceso es igual a la posición inicial del siguiente proceso quiere decir que no hay hueco debajo
167  else {
168  peso_hueco = it->first - (pos + peso); //posición inicial siguiente proceso - posición final proceso será el tamaño del hueco
169  pos_hueco = pos + peso;
170  set<pair<int, int>, ComparePairs>::iterator it2 = huecos.find(make_pair(peso_hueco, pos_hueco));
171  huecos.erase(it2);
172  return true;
173  }
174  }
175  else {
176  if (pos + peso == memtotal) return false;
177  else { //si posición final del proceso es distinta a la memoria total quiere decir que hay hueco abajo
178  pos_hueco = pos + peso;
179  peso_hueco = memtotal - pos_hueco;
180  set<pair<int, int>, ComparePairs>::iterator it2 = huecos.find(make_pair(peso_hueco, pos_hueco));
181  huecos.erase(it2);
